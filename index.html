      function setSplit(pct){

        const clamped = Math.max(0, Math.min(100, pct));

        after.style.clipPath = `inset(0 0 0 ${clamped}%)`;

        handle.style.left = `${clamped}%`;

        knob.style.left = `${clamped}%`;

      }

      // начальное значение

      const init = parseInt(ba.getAttribute('data-default') || '50', 10);

      setSplit(init);

      range.addEventListener('input', e => setSplit(e.target.value));

      // мышь/тач по области

      function updateFromEvent(e){

        const rect = ba.getBoundingClientRect();

        const x = (e.touches ? e.touches[0].clientX : e.clientX) - rect.left;

        setSplit( (x / rect.width) * 100 );

      }

      ba.addEventListener('pointerdown', e => { ba.setPointerCapture(e.pointerId); updateFromEvent(e) });

      ba.addEventListener('pointermove', e => { if (e.pressure>0 || e.buttons) updateFromEvent(e) });

      ba.addEventListener('pointerup', e => { /* noop */ });

    });
